\textbf{\Large Presentaci\'o Probabilitat i Processos Aleatoris (PIPA)}

\vskip 1 cm
\noindent
\textbf{Prerrequisits:}
\renewcommand{\labelenumi}{\m@th\blacktriangleright}
\renewcommand{\labelenumiii}{\m@th\bullet}
\begin{itemize}
\item FMII (Probabilitat):
\begin{itemize}
\item	conceptes bàsics: experiment aleatori, espai mostral, succés elemental,  -àlgebra, mesura de probabilitat (clàssica, freqüencialista, axiomàtica)
\item	combinatòria (variacions amb i sense repetició, combinacions, permutacions amb repetició)
\item	propietats bàsiques:
\begin{itemize}
\item	$P(A \cup B) = P(A) + P(B) - P(A \cap B)$
\item	$P(A \cap B) = P(A|_B) P(B)$
\item	Si $A$ i $B$ són independents: $P(A|_B) = P(A)$, llavors $P(A \cap B)=P(A)P(B)$
\item	Si $A$ i $B$ són disjunts, $A \cap  B = \emptyset$ , llavors $P(A \cup B) = P(A) + P(B)$
\item	Regla de Bayes: 
\[
P(A|_B)=\frac{P(A \cap B)}{P(B)}
\]
\item	Fòrmula de la Probabilitat Total:
\[
P(A)=P(A|_{B_1}) P(B_1) + P(A|_{B_2}) P(B_2) + \cdots + P(A|_{B_n}) P(B_n)
\]
\noindent		
on $B_1, B_2, \cdots, B_n$ formen una partició de l'espai mostral
\end{itemize}
\item	variables aleatòries:
\begin{itemize}
\item	definició: funció que associa els elements d'un espai mostral amb els nombres reals, 
de manera que podem calcular la probabilitat de qualsevol interval de nombres reals
\item	v.a. discretes: 
\begin{itemize}
\item	funció de probabilitat
\item	funció de distribució
\item	v.a. discretes típiques: Bernouilli, binomial, Poisson
\end{itemize}
\item	v.a. contínues:
\begin{itemize}
\item	funció de distribució
\item funció de densitat
\item	v.a. contínues típiques: uniforme, exponencial, gaussiana
\end{itemize}
\item	transformacions (funcions) de variables aleatòries discretes i contínues
\end{itemize}
\item	moments de variables aleatòries contínues i discretes:
\begin{itemize}
\item	esperança
\item	variància
\item	moments de funcions de v.a.
\item	desigualtats de Markov i Txebitxef
\end{itemize}
\end{itemize}
\item	FMI (Integració):
\begin{itemize}
\item	integral dobles (es suposa que les integrals simples ja es dominen)
\end{itemize}
\end{itemize}
